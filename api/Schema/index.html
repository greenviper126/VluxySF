interface SoundSchema {

A folder for storing unattached sounds.

SoundGroupContainer: Configuration--

A folder for storing sound groups.

SoundGroups: SoundGroups--

A dictionary of sound group instances, organized by name.

SoundDefinitions: SoundDefinitions--

A dictionary of sound definitions, organized by sound name.

DefinitionsGroupHash: Dictionary<Dictionary<string>>--

Set of hashmaps for fast searching.

PreloadIds: {string}--

A list of sound asset IDs to preload.

}
